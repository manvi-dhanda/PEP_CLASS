Ques :Input 2 nos. as per user perform the perations like add, sub,multi,divide,left shift,right shift etc
in CPP

Solu: #include <iostream>
using namespace std;
int main()
{
    string op;
    int a,b;
    cin>>op>>a>>b;
    if(op="add"){
        cout<<(a+b)<<endl;
    }
    if(op="sub")
    {
        cout<<(a-b)<<endl;
    }
    if(op="multi")
    {
        cout<<(a*b)<<endl;
    }
    if(op="divide")
    {
        cout<<(a/b)<<endl;
    }
    if(op="bsr")
    {
        cout<<(a>>b)<<endl;
    }
    if(op="bsl")
    {
        cout<<(a<<b)<<endl;
    }
}
